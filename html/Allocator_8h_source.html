201  pointer allocate_if_possible(size_t bytes_read, int current_sentinel,
202  int bytes_needed, size_t n)
203  {
204  // If there is enough space in this block, allocate:
205  if (current_sentinel >= bytes_needed && current_sentinel > 0)
206  {
207  // Calculate the remaining space there would be in this block if
208  // n Ts have been allocated
209  int remaining_space = current_sentinel + (2 * sizeof(int)) -
210  ((2 * sizeof(int)) + (n * sizeof(T)));
211 
212  // If there is enough space to allocate n Ts AND another block
213  // that is AT LEAST bigger than the "smallest allowable block",
214  // create two more sentinels to designate the latter
215  if (remaining_space >= (2 * sizeof(int) + sizeof(T)))
216  {
217  // Create pointer to beginning of allocated space
218  bytes_read += sizeof(int);
219  pointer p = reinterpret_cast<pointer>(&a[bytes_read]);
220 
221  // Modify first sentinel to new value
222  (*this)[bytes_read - sizeof(int)] = bytes_needed * -1;
223 
224  // Go to second sentinel
225  bytes_read += bytes_needed;
226 
227  // Create an end sentinel
228  (*this)[bytes_read] = bytes_needed * -1;
229 
230  // Go to first in second group of sentinels
231  bytes_read += sizeof(int);
232 
233  // Create first sentinel in second group
234  (*this)[bytes_read] = remaining_space - 2 * sizeof(int);
235 
236  // Create second sentinel in second group
237  bytes_read += remaining_space - sizeof(int);
238 
239  (*this)[bytes_read] = remaining_space - 2 * sizeof(int);
240 
241  assert(valid());
242  return p;
243  }
244  // If there's not enough space for another T and 2 sentinels,
245  // just fill in this block
246  else
247  {
248  pointer p = reinterpret_cast<pointer>(&a[bytes_read + sizeof(int)]);
249 
250  // Mark current pair of sentinels as "used"
251  (*this)[bytes_read] *= -1;
252  (*this)[bytes_read + current_sentinel + sizeof(int)] *= -1;
253 
254  assert(valid());
255  return p;
256  }
257  }
258 
259  // Return null pointer if this block was not suitable
260  return nullptr;
261  }

395  {
396  if (p == 0)
397  {
398  throw std::invalid_argument("Invalid p pointer");
399  }
400 
401  // Get the value of the sentinel of the block that p is pointing to
402  int* sentinel_pointer = reinterpret_cast<int*>(p) - 1;
403  int sentinel_value = *(sentinel_pointer);
404 
405  // Find the beginning and end of a
406  int* beginning_of_a = reinterpret_cast<int*>(&a);
407  int* end_of_a = reinterpret_cast<int*>(&a[sizeof(a) / sizeof(a[0])]);
408 
409  // Find the last valid location that a sentinel can be at
410  char* end_minus_block = reinterpret_cast<char*>(end_of_a) -
411  (2 * sizeof(int) - sizeof(T));
412  int* last_valid_location = reinterpret_cast<int*>(end_minus_block);
413 
414  // Make sentinel_value positive
415  if(sentinel_value < 0)
416  {
417  sentinel_value *= -1;
418  }
419 
420  // Mark the "beginning sentinel" in this block
421  int* first_sentinel = sentinel_pointer;
422  char* reader = reinterpret_cast<char*>(first_sentinel);
423 
424  // Mark the "end sentinel" in this block
425  char* second_reader = reinterpret_cast<char*>(first_sentinel) +
426  sentinel_value + sizeof(int);
427  int* end_sentinel = reinterpret_cast<int*>(second_reader);
428 
429  // Check for validity before you begin actual deallocation
430  if(*(end_sentinel) > 0 ||
431  (*(end_sentinel) * -1) != sentinel_value ||
432  reinterpret_cast<int*>(p) < beginning_of_a ||
433  reinterpret_cast<int*>(p) > last_valid_location)
434  {
435  throw std::invalid_argument("Invalid p pointer");
436  }
437 
438  // Check for free blocks BEFORE this block
439  if(first_sentinel - 1 > beginning_of_a)
440  {

442  int previous_sentinel_value = *(reinterpret_cast<int*>(reader - sizeof(int)));
443  if(previous_sentinel_value > 0)
444  {
445 
446  // Set the first_sentinel to the previous block's first sentinel
447  first_sentinel = reinterpret_cast<int*>(reader - previous_sentinel_value - (2 * sizeof(int)));
448 
449  // Accumulate sentinel value
450  sentinel_value += previous_sentinel_value + (2 * sizeof(int));
451  }
452  }
453

455  if(end_sentinel + 1 < end_of_a)
456  {
457  // Check if next block is free
